.example-10 .animated-grid {
  display: grid;
  grid-template-rows: 0fr;
  transition: grid-template-rows 0.2s ease;
}

/* aminate on hover */
.example-10 .btn.show_hover:hover ~ .animated-grid {
  grid-template-rows: 1fr;
}

/*class to toggle with JS */
.example-10 .animated-grid.expand_rows {
  grid-template-rows: 1fr;
}

.item-1 {
  overflow: hidden;
}
const example10 = () => {
const toggle_button_title = document.querySelector('.toggle_click > span')
toggle_button_title.textContent !== 'Click to hide' ? toggle_button_title.textContent = 'Click to hide'
    : toggle_button_title.textContent = 'Click to show';
document.querySelector('.animated-grid').classList.toggle('expand_rows');
  }
}
